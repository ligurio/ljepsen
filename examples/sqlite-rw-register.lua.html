sqlite-rw-register.lua

local sqlite3 = require('lsqlite3')
local ljepsen = require('ljepsen')
local os = require('os')

local client = ljepsen.client
local runner = ljepsen.runner
local rw_register_gen = ljepsen.tests.rw_register_gen

print('SQLite version:', sqlite3.version())
print('lsqlite3 library version:', sqlite3.lversion())

local db = sqlite3.open_memory() -- Open SQLite database in memory.

if os.getenv('DEV') == 'ON' then
    db:trace(function(_, sql)
        print('SQLite trace:', sql)
    end)
end

assert(sqlite3.OK == db:exec('CREATE TABLE rw_register (id, val)'))

local sqlite_rw_register = client.new()

local sqlite_insert_stmt = assert(db:prepare('INSERT INTO rw_register VALUES (?, ?)'))

local function insert(key, val)
    local ok = sqlite_insert_stmt:bind_values(key, val)
    if ok ~= sqlite3.OK then
        return false
    end
    ok = sqlite_insert_stmt:step()
    if ok ~= sqlite3.DONE then
        return false
    end
    sqlite_insert_stmt:reset()

    return true
end

local OP_TYPE = 1
local OP_VAL = 3

local KEY_ID = 1

-- sqlite_rw_register implements a client that performs two operations on
-- database: read and write. Method invoke must apply these operations to a
-- database instance.
sqlite_rw_register.invoke = function(self, op)
    local val = op.value[1]
    local type = 'ok'
    if val[OP_TYPE] == 'r' then
        val[OP_VAL] = db:exec(string.format('SELECT val FROM rw_register WHERE id = %d', KEY_ID))
    elseif val[OP_TYPE] == 'w' then
        local ok = insert(KEY_ID, val[OP_VAL])
        if ok == false then
            type = 'fail'
        end
    else
        error('Unknown operation')
    end

    return {
        value = { val },
        f = op.f,
        process = op.process,
        type = type,
    }
end

local test_options = {
    create_reports = true,
    threads = 5,
}
local ok = runner.run_test({
    client = sqlite_rw_register,
    generator = rw_register_gen(1000),
}, test_options)

if not ok then
    print('Test has failed.')
end

sqlite_insert_stmt:finalize()

local changes = db:total_changes()
assert(changes == 504, string.format('Number of operations is wrong (%d != 504)', changes))
print('Total changes in SQLite DB:', changes)

-- Teardown.
db:close()

if os.getenv('DEV') ~= 'ON' then
    os.remove('history.json')
    os.remove('history.txt')
end
