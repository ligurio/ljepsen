sqlite-list-append.lua

local sqlite3 = require('lsqlite3')
local ljepsen = require('ljepsen')
local math = require('math')
local os = require('os')

local client = ljepsen.client
local gen_lib = ljepsen.gen
local runner = ljepsen.runner

-- A partial test, including a generator and checker. Youâ€™ll need to provide a
-- client which can understand operations of the form:
--
-- {:type :invoke, :f :txn, :value [[:r 3 nil] [:append 3 2] [:r 3]]}
-- and return completions like:
-- {:type :invoke, :f :txn, :value [[:r 3 [1]] [:append 3 2] [:r 3 [1 2]]]}

--[[
Values are lists of integers. Each operation performs a transaction,
comprised of micro-operations which are either reads of some value (returning
the entire list) or appends (adding a single number to whatever the present
value of the given list is). We detect cycles in these transactions using
Jepsen's cycle-detection system.
]]

local function op_read()
    return {
        f = 'txn',
        value = {
            {
                'r',
                1,
                nil,
            }
        },
    }
end

local function op_append()
    return {
        f = 'txn',
        value = {{
            'append',
            1,
            math.random(1, 100),
        }}
    }
end

local db = sqlite3.open_memory() -- Open SQLite database in memory.
db:exec[[ CREATE TABLE IF NOT EXISTS test (id INT NOT NULL PRIMARY KEY, sk INT NOT NULL, val TEXT) ]]

local sqlite_list_append = client.new()

--[[

CREATE TABLE IF NOT EXISTS t3 (id INT NOT NULL PRIMARY KEY, sk INT NOT NULL, val TEXT);
UPDATE t3 SET val = 3 WHERE id = 3;
SELECT val FROM t3 WHERE id = 3;
INSERT INTO t3 (id, sk, val) VALUES (3, 3, 3);

]]

-- Description in ./yugabyte/src/yugabyte/ysql/append_table.clj

sqlite_list_append.invoke = function(self, op)
    -- [[:r 3 nil] [:append 3 2] [:r 3]]
    local val = op.value[1]
    print(require('log').info(val))
    local key = val[2]
    local op_type = val[2]
    print('key and op', key, op)
    if op_type == 'r' then
        val[3] = db:exec(string.format('SELECT val FROM test WHERE id = %d', key))
    elseif op_type == 'append' then
        db:exec(string.format('UPDATE test SET val = %d WHERE id = %d', val[3], key))
        db:exec(string.format('INSERT INTO test VALUES (%d, %d)', key, val[3]))
    else
        error('Unknown operation')
    end

    return {
        value = { val },
        f = op.f,
        process = op.process,
    }
end

local test_options = {
    create_reports = true,
    threads = 1,
}
local ok = runner.run_test({
    client = sqlite_list_append,
    generator = gen_lib.cycle(gen_lib.iter({
        op_read, op_append
    })):take(5)
}, test_options)

if ok == true then
    print('Feel free to check history.json with elle-cli:')
    print('java -jar elle-cli-0.1.1-standalone.jar --model elle-list-append history.json')
end

local changes = db:total_changes()
assert(changes == 2, string.format('Number of operations is wrong (%d vs 2)', changes))
print('Total changes in SQLite DB:', changes)
db:close() -- Close a database handler.

if os.getenv('DEV') ~= 'ON' then
    os.remove('history.json')
    os.remove('history.txt')
end
