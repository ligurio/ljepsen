sqlite-list-append.lua

local sqlite3 = require('lsqlite3')
local ljepsen = require('ljepsen')
local os = require('os')

local list_append_gen = ljepsen.tests.list_append_gen
local client = ljepsen.client
local runner = ljepsen.runner

--[[
Values are lists of integers. Each operation performs a transaction,
comprised of micro-operations which are either reads of some value (returning
the entire list) or appends (adding a single number to whatever the present
value of the given list is). We detect cycles in these transactions using
Jepsen's cycle-detection system.

Lists are encoded as rows in a table; key names are table names, and the set
of all rows determines the list contents.

This test requires a way to order table contents.
]]

local db = sqlite3.open_memory() -- Open SQLite database in memory.

print('SQLite version:', sqlite3.version())
print('lsqlite3 library version:', sqlite3.lversion())

if os.getenv('DEV') == 'ON' then
    db:trace(function(_, sql)
        print('SQLite trace:', sql)
    end)
end

local sqlite_list_append = client.new()

assert(sqlite3.OK == db:exec('CREATE TABLE list_append (key INT NOT NULL, val INT)'))

local sqlite_insert_stmt = assert(db:prepare('INSERT INTO list_append VALUES (?, ?)'))

local function insert(key, val)
    sqlite_insert_stmt:bind_values(key, val)
    sqlite_insert_stmt:step()
    sqlite_insert_stmt:reset()
end

local sqlite_select_stmt = assert(db:prepare('SELECT key, val FROM list_append ORDER BY key'))

local function select(key)
    local t = {}
    for row in sqlite_select_stmt:nrows() do
        if row.key == key then
            table.insert(t, row.val)
        end
    end

    return t
end

local MOP_TYPE = 1
local MOP_KEY = 2
local MOP_VAL = 3

sqlite_list_append.invoke = function(self, op)
    local mop = op.value[1] -- TODO: Support more than one mop in operation.
    local mop_key = mop[MOP_KEY]
    local type = 'ok'
    if mop[MOP_TYPE] == 'r' then
        mop[MOP_VAL] = select(mop_key)
    elseif mop[MOP_TYPE] == 'append' then
        insert(mop_key, mop[MOP_VAL])
    else
        error('Unknown operation')
    end

    return {
        value = { mop },
        f = op.f,
        process = op.process,
        type = type,
    }
end

local test_options = {
    create_reports = true,
    threads = 1,
}
local ok = runner.run_test({
    client = sqlite_list_append,
    generator = list_append_gen(100)
}, test_options)

if not ok then
    print('Test has failed.')
end

local changes = db:total_changes()
assert(changes == 50, string.format('Number of operations is wrong (%d vs 50)', changes))
print('Total changes in SQLite DB:', changes)
db:close() -- Close a database handler.

if os.getenv('DEV') ~= 'ON' then
    os.remove('history.json')
    os.remove('history.txt')
end
